def Board.move(piece, dir):
    copy = piece.copy()
    copy.move(dir)
    moveTo(piece, copy)

def Board.rotate(piece, dir):
    copy = piece.copy()
    copy.rotate(dir)
    moveTo(piece, copy)

def Board.moveTo(piece, copy):
    board.remove(piece)

    if (board.detectCollision(copy))
        board.addPiece(piece)
        copy.dispose()
        return false;

    board.addPiece(copy)
    // como fazer a troca corretamente?
    piece = copy
    piece.dispose()
    return true
